    <script type="importmap">
        {
            "imports": {
                "three": "https://unpkg.com/three@0.160.0/build/three.module.js",

    <script type="module">
        import * as THREE from 'three';
        import { OrbitControls } from 'three/addons/controls/OrbitControls.js';
        import { PLYLoader } from 'three/addons/loaders/PLYLoader.js';
        import { InsectSwarm } from '../agents/swarm.js';
        import { TerrestrialClass } from '../agents/terrestrial.js';
        import { EndangeredClass } from '../agents/endangered.js'; 
        import { createBeeVisionMaterial } from '../shaders/beeVision.js';
        import { createTerrestrialVisionMaterial } from '../shaders/terrestrialVision.js'; 
        import { createGiantVisionMaterial } from '../shaders/giantVision.js'; 

        const scene = new THREE.Scene();
        scene.fog = new THREE.FogExp2(0x05050A, 0.003);

        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 2.5, 1000);
        camera.position.set(0, 30, 80);

        const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(window.devicePixelRatio);
        document.body.appendChild(renderer.domElement);

        const controls = new OrbitControls(camera, renderer.domElement);
        controls.enableDamping = true;
        controls.dampingFactor = 0.05; 
        controls.autoRotate = false; 
        controls.enableZoom = false; 
        controls.rotateSpeed = -1.0; 

        renderer.domElement.addEventListener('wheel', (event) => {
            if (cameraMode === "tethered" && activeLens !== "human") return; 
            event.preventDefault(); 
            const direction = new THREE.Vector3().subVectors(camera.position, controls.target);
            let currentDistance = direction.length();
            const zoomAmount = (event.deltaY > 0) ? 5 : -5;
            let newDistance = currentDistance + zoomAmount;
            newDistance = Math.max(10, Math.min(250, newDistance)); 
            direction.normalize().multiplyScalar(newDistance);
            camera.position.copy(controls.target).add(direction);
        }, { passive: false });

        let activeLens = "human";
        let cameraMode = "god";
        let humanMaterial; 
        let environmentPoints;
        let originalEnvPositions;
        let isTethered = false;
        let previousAgentPos = new THREE.Vector3();
        const clock = new THREE.Clock(); 

        const beeMaterial = createBeeVisionMaterial();
        const terrestrialMaterial = createTerrestrialVisionMaterial(); 
        const giantMaterial = createGiantVisionMaterial();

        const beeSwarm = new InsectSwarm(scene, 500);
        const groundSwarm = new TerrestrialClass(scene, 150);
        const giantSwarm = new EndangeredClass(scene, 5); 

        let audioCtx, masterGain, beeGain, groundGain;
        let audioInitialized = false;
        let audioMuted = false; 
        let lastPingTime = 0;

        const btnAudio = document.getElementById('btn-audio');
        btnAudio.addEventListener('click', () => {
            if (!audioInitialized) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                masterGain = audioCtx.createGain();
                masterGain.connect(audioCtx.destination);
                masterGain.gain.value = 0.5;

                beeGain = audioCtx.createGain();
                beeGain.gain.value = 0; 
                const beeOsc1 = audioCtx.createOscillator();
                const beeOsc2 = audioCtx.createOscillator();
                beeOsc1.type = 'sawtooth'; beeOsc2.type = 'sawtooth';
                beeOsc1.frequency.value = 120; beeOsc2.frequency.value = 124; 
                beeOsc1.connect(beeGain); beeOsc2.connect(beeGain);
                beeGain.connect(masterGain);
                beeOsc1.start(); beeOsc2.start();

                groundGain = audioCtx.createGain();
                groundGain.gain.value = 0;
                const groundOsc = audioCtx.createOscillator();
                groundOsc.type = 'sine';
                groundOsc.frequency.value = 45; 
                groundOsc.connect(groundGain);
                groundGain.connect(masterGain);
                groundOsc.start();

                audioInitialized = true;
                btnAudio.innerText = "[ AUDIO SYSTEM: ONLINE ]";
                btnAudio.style.animation = "none";
                btnAudio.style.color = "#FFFFFF";
                btnAudio.style.borderColor = "#FFFFFF";
            } else {
                audioMuted = !audioMuted;
                if (audioMuted) {
                    audioCtx.suspend(); 
                    btnAudio.innerText = "[ AUDIO SYSTEM: MUTED ]";
                    btnAudio.style.color = "#888888";
                    btnAudio.style.borderColor = "#888888";
                } else {
                    audioCtx.resume(); 
                    btnAudio.innerText = "[ AUDIO SYSTEM: ONLINE ]";
                    btnAudio.style.color = "#FFFFFF";
                    btnAudio.style.borderColor = "#FFFFFF";
                }
            }
        });

        function triggerSonarPing() {
            if (!audioInitialized || audioMuted) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.type = 'sine';
            osc.frequency.setValueAtTime(800, audioCtx.currentTime);
            osc.frequency.exponentialRampToValueAtTime(300, audioCtx.currentTime + 0.5);
            gain.gain.setValueAtTime(0.6, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 1.0);
            osc.connect(gain);
            gain.connect(masterGain);
            osc.start();
            osc.stop(audioCtx.currentTime + 1.0);
        }

        // --- CONTEXT-AWARE RECALL LOGIC ---
        let recallActive = false;
        const btnRecall = document.getElementById('btn-recall');
        
        function updateRecallTargets() {
            if (!recallActive) {
                // If it's offline, shut everyone down
                beeSwarm.setRecallMode(false);
                groundSwarm.setRecallMode(false);
                giantSwarm.setRecallMode(false);
                return;
            }
            // If it's online, ONLY call the ones we are currently looking at
            beeSwarm.setRecallMode(activeLens === "human" || activeLens === "bee");
            groundSwarm.setRecallMode(activeLens === "human" || activeLens === "terrestrial");
            giantSwarm.setRecallMode(activeLens === "human" || activeLens === "giant");
        }

        btnRecall.addEventListener('click', () => {
            recallActive = !recallActive;
            updateRecallTargets();
            
            if (recallActive) {
                btnRecall.innerText = "[ GENESIS RECALL: ONLINE ]";
                btnRecall.style.color = "#FFFFFF";
                btnRecall.style.borderColor = "#FFFFFF";
                btnRecall.style.animation = "none";
            } else {
                btnRecall.innerText = "[ GENESIS RECALL: OFFLINE ]";
                btnRecall.style.color = "#FFB300";
                btnRecall.style.borderColor = "#FFB300";
                btnRecall.style.animation = "blink 1s step-end infinite";
            }
        });

        const btnLensHuman = document.getElementById('lens-human');
        const btnLensBee = document.getElementById('lens-bee');
        const btnLensGround = document.getElementById('lens-ground');
        const btnLensGiant = document.getElementById('lens-giant');
        const btnCamGod = document.getElementById('cam-god');
        const btnCamTether = document.getElementById('cam-tether');

        const hudLayer = document.getElementById('hud-layer');
        const hudHeader = document.getElementById('hud-header');
        const hudId = document.getElementById('hud-id');
        const hudAlt = document.getElementById('hud-alt');
        const hudVel = document.getElementById('hud-vel');
        const hudStatus = document.getElementById('hud-status');
        const hudNrg = document.getElementById('hud-nrg'); 

        function updateUI() {
            btnLensHuman.style.background = "transparent"; btnLensHuman.style.color = "#00FFFF";
            btnLensBee.style.background = "transparent"; btnLensBee.style.color = "#FFB300";
            btnLensGround.style.background = "transparent"; btnLensGround.style.color = "#FF0055";
            btnLensGiant.style.background = "transparent"; btnLensGiant.style.color = "#9D00FF";

            if(activeLens === "human") { btnLensHuman.style.background = "#00FFFF"; btnLensHuman.style.color = "#000"; }
            if(activeLens === "bee") { btnLensBee.style.background = "#FFB300"; btnLensBee.style.color = "#000"; }
            if(activeLens === "terrestrial") { btnLensGround.style.background = "#FF0055"; btnLensGround.style.color = "#000"; }
            if(activeLens === "giant") { btnLensGiant.style.background = "#9D00FF"; btnLensGiant.style.color = "#000"; }

            btnCamGod.style.background = "transparent"; btnCamGod.style.color = "#FFFFFF";
            btnCamTether.style.background = "transparent"; btnCamTether.style.color = "#FFFFFF";

            if(cameraMode === "god") { btnCamGod.style.background = "#FFFFFF"; btnCamGod.style.color = "#000"; }
            if(cameraMode === "tethered") { btnCamTether.style.background = "#FFFFFF"; btnCamTether.style.color = "#000"; }

            if (cameraMode === "tethered") {
                hudLayer.style.display = "block";
                let themeColor = "#00FFFF"; 
                if(activeLens === "bee") themeColor = "#FFB300";
                if(activeLens === "terrestrial") themeColor = "#FF0055";
                if(activeLens === "giant") themeColor = "#9D00FF";

                hudLayer.style.borderColor = themeColor;
                hudLayer.style.boxShadow = `0 0 20px ${themeColor}40`; 
                hudHeader.style.color = themeColor;
                hudHeader.style.borderBottomColor = themeColor;
                hudId.style.color = themeColor;
                hudStatus.style.color = themeColor;
            } else {
                hudLayer.style.display = "none";
            }

            if (audioInitialized) {
                if (activeLens === "human" || activeLens === "giant") {
                    beeGain.gain.setTargetAtTime(0.001, audioCtx.currentTime, 0.1);
                    groundGain.gain.setTargetAtTime(0.001, audioCtx.currentTime, 0.1);
                } else if (activeLens === "terrestrial") {
                    beeGain.gain.setTargetAtTime(0.001, audioCtx.currentTime, 0.1);
                    groundGain.gain.setTargetAtTime(0.5, audioCtx.currentTime, 0.5); 
                }
            }
        }

        function applyLens() {
            if(!environmentPoints) return;
            if(activeLens === "human") environmentPoints.material = humanMaterial;
            if(activeLens === "bee") environmentPoints.material = beeMaterial;
            if(activeLens === "terrestrial") environmentPoints.material = terrestrialMaterial;
            if(activeLens === "giant") environmentPoints.material = giantMaterial;
            updateUI();
            
            // If we swap lenses while recall is active, update who is being recalled!
            updateRecallTargets(); 
        }

        btnLensHuman.addEventListener('click', () => { activeLens = "human"; applyLens(); });
        btnLensBee.addEventListener('click', () => { activeLens = "bee"; applyLens(); });
        btnLensGround.addEventListener('click', () => { activeLens = "terrestrial"; applyLens(); });
        btnLensGiant.addEventListener('click', () => { activeLens = "giant"; applyLens(); });

        btnCamGod.addEventListener('click', () => { cameraMode = "god"; updateUI(); });
        btnCamTether.addEventListener('click', () => { cameraMode = "tethered"; updateUI(); });

        updateUI();

        const loader = new PLYLoader();
        loader.load('../seeds/pandora_seed_001.ply', function (geometry) {
            const circleCanvas = document.createElement('canvas');
            circleCanvas.width = 32;
            circleCanvas.height = 32;
            const ctx = circleCanvas.getContext('2d');
            ctx.beginPath();
            ctx.arc(16, 16, 16, 0, Math.PI * 2);
            ctx.fillStyle = '#ffffff';
            ctx.fill();
            const circleTexture = new THREE.CanvasTexture(circleCanvas);
            
            const material = new THREE.PointsMaterial({
                size: 0.4, 
                vertexColors: true,
                transparent: true,
                opacity: 0.8,
                map: circleTexture, 
                alphaTest: 0.5      
            });

            humanMaterial = material; 
            environmentPoints = new THREE.Points(geometry, humanMaterial);
            scene.add(environmentPoints);

            originalEnvPositions = new Float32Array(geometry.attributes.position.array);
        });

        function animate() {
            requestAnimationFrame(animate);
            const elapsedTime = clock.getElapsedTime();
            
            beeMaterial.uniforms.time.value = elapsedTime;
            terrestrialMaterial.uniforms.time.value = elapsedTime;
            giantMaterial.uniforms.time.value = elapsedTime;

            beeSwarm.update(); 
            groundSwarm.update(); 
            giantSwarm.update(); 

            if (audioInitialized && !audioMuted) {
                if (activeLens === 'bee') {
                    const dist = camera.position.length();
                    const targetVol = Math.max(0.001, 0.2 - (dist / 750)); 
                    beeGain.gain.setTargetAtTime(targetVol, audioCtx.currentTime, 0.1);
                }
                if (activeLens === 'giant') {
                    if (elapsedTime - lastPingTime > 2.5) {
                        triggerSonarPing();
                        lastPingTime = elapsedTime;
                    }
                }
            }

            if (cameraMode === "tethered") {
                let currentAgentPos;

                if (activeLens === "human") {
                    const walkRadius = 70;
                    const walkSpeed = 0.02; 
                    currentAgentPos = new THREE.Vector3(
                        Math.cos(elapsedTime * walkSpeed) * walkRadius,
                        2.0, 
                        Math.sin(elapsedTime * walkSpeed) * walkRadius
                    );
                    
                    hudId.innerText = "OBSERVER_GHOST";
                    hudStatus.innerText = "ECOLOGICAL_SURVEY";
                    hudNrg.innerText = "---";
                    
                } else {
                    let swarmArray;
                    
                    if (activeLens === "terrestrial") {
                        swarmArray = groundSwarm.points.geometry.attributes.position.array;
                        hudId.innerText = "SCAVENGER_UNIT_001";
                        hudStatus.innerText = groundSwarm.states[ 0 ]; 
                        hudNrg.innerText = Math.max(0, Math.floor(groundSwarm.energies[ 0 ])); 
                    } else if (activeLens === "giant") {
                        swarmArray = giantSwarm.points.geometry.attributes.position.array;
                        hudId.innerText = "PILGRIM_PRIME";
                        hudStatus.innerText = giantSwarm.states[ 0 ]; 
                        hudNrg.innerText = Math.max(0, Math.floor(giantSwarm.energies[ 0 ])); 
                    } else { 
                        swarmArray = beeSwarm.points.geometry.attributes.position.array;
                        hudId.innerText = "INSECT_UNIT_001";
                        hudStatus.innerText = beeSwarm.states[ 0 ]; 
                        hudNrg.innerText = Math.max(0, Math.floor(beeSwarm.energies[ 0 ])); 
                    }
                    
                    currentAgentPos = new THREE.Vector3(swarmArray[ 0 ], swarmArray[ 1 ], swarmArray[ 2 ]);
                }

                if (!isTethered) {
                    camera.position.copy(currentAgentPos);
                    controls.target.set(currentAgentPos.x, currentAgentPos.y, currentAgentPos.z - 1);
                    isTethered = true;
                } else {
                    const movementDelta = new THREE.Vector3().subVectors(currentAgentPos, previousAgentPos);
                    camera.position.add(movementDelta);
                    controls.target.add(movementDelta);
                    
                    const speed = movementDelta.length() * 60; 
                    hudAlt.innerText = currentAgentPos.y.toFixed(2);
                    hudVel.innerText = speed.toFixed(2);
                }
                
                previousAgentPos.copy(currentAgentPos);
                
            } else {
                isTethered = false;
                controls.target.lerp(new THREE.Vector3(0, 0, 0), 0.05);
            }
            
            controls.update(); 

            if (environmentPoints && giantSwarm) {
                const envPos = environmentPoints.geometry.attributes.position.array;
                const giantPos = giantSwarm.points.geometry.attributes.position.array;
                
                for(let i = 0; i < envPos.length; i += 3) {
                    let originX = originalEnvPositions[ i ];
                    let originY = originalEnvPositions[ i + 1 ];
                    let originZ = originalEnvPositions[ i + 2 ];
                    
                    if (originY > 25) continue;
                    
                    let dx = 0; let dz = 0;
                    
                    for(let g = 0; g < giantSwarm.count; g++) {
                        let gIndex = g * 3;
                        let gx = giantPos[ gIndex ]; let gz = giantPos[ gIndex + 2 ];
                        
                        let distSq = (originX - gx)*(originX - gx) + (originZ - gz)*(originZ - gz);
                        let repulseRadius = 8; let radiusSq = repulseRadius * repulseRadius;
                        
                        if (distSq < radiusSq) {
                            let force = (radiusSq - distSq) / radiusSq; 
                            dx += (originX - gx) * force * 0.3; 
                            dz += (originZ - gz) * force * 0.3; 
                        }
                    }
                    
                    envPos[ i ] = originX + dx; envPos[ i + 2 ] = originZ + dz;
                }
                environmentPoints.geometry.attributes.position.needsUpdate = true;
            }
            renderer.render(scene, camera);
        }
        animate();

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });
